Esse projeto mostra os número pares, ímpares e primos de determinado intervalo

Esse código foi transcrito de python para javascript para se adequar a este projeto.

Código em JS:


    function even(start, end) {
        const nums = new Array();
    
        for (let i = start; i <= end; i++) {
            nums.push(i);
        }
    
        const evenNumbers = nums.filter(value => {
            if (value % 2 === 0) return value;
        });
    
        return evenNumbers;
    }
    
    function odd(start, end) {
        const nums = new Array();
    
        for (let i = start; i <= end; i++) {
            nums.push(i);
        }
    
        const oddNumbers = nums.filter(value => {
            if (value % 2 !== 0) return value;
        });
    
        return oddNumbers;
    }
    
    function primeNumber(start, end) {
        const nums = new Array();
    
        for (let i = start; i <= end; i++) {
            nums.push(i);
        }
    
        const primeNumbers = nums.filter(value => {
            let count = 0;
    
            for (let i = 1; i <= value; i++) {
                if (value % i === 0) count++;
            }
            if (count <= 2) return value
    
        });
    
        return primeNumbers;
    }

Código em Python:


    def even(start, end):
        nums = [n for n in range(start, end + 1)]
        even_numbers = []
        for n in nums:
            if n % 2 == 0:
                even.append(n)

        return even_numbers


    def odd(start, end):
        nums = [n for n in range(start, end + 1)]
        odd_numbers = []
        for n in nums:
            if n % 2 != 0:
                odd.append(n)

        return odd_numbers


    def prime_number(start, end):
        nums = [n for n in range(start, end + 1)]
        prime_numbers = []
        for n in nums:
            count = 0
            for i in range(1, (n + 1)):
                if n % i == 0:
                    count += 1

            if count <= 2:
                prime_numbers.append(n)

        return prime_numbers
